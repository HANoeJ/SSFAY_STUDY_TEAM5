import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.lang.reflect.Array;
import java.util.ArrayDeque;
import java.util.Arrays;
import java.util.Deque;

public class Main {
  static int N;
  static int[][] results;
  static int ans = -1;
  static void permutation(int idx, int[] sel, boolean[] visited){
    if(idx == sel.length){
//      System.out.println(Arrays.toString(sel));
      run(sel.clone());
      return;
    }
    for(int i = 0 ; i < 8 ; i++){
      if(visited[i]) continue;
      visited[i] = true;
      sel[idx] = i;
      permutation(idx + 1, sel, visited);
      visited[i] = false;
    }
  }
  static void run(int[] order){
    int[] pOrder = new int[9];
    for(int i = 0 ; i < 3 ; i ++){
      pOrder[i] = order[i] + 1;
    }
    for(int i = 4 ; i < 9 ; i ++){
      pOrder[i] = order[i - 1] + 1;
    }
    int out = 0;
    int inning = 0;
    int playerN = 0;
    int score = 0;
    int[] ground = new int[3];
    while(inning < N){
      int player = pOrder[playerN];
      int result = results[inning][player];
      if(result == 1){
        int tmp = ground[2];
        for(int i = 2 ; i > 0 ; i--){
          ground[i] = ground[i - 1];
        }
        ground[0] = 1;
        score += tmp;
      }
      else if(result == 2){
        int tmp = ground[2] + ground[1];
        for(int i = 2 ; i > 0 ; i--){
          ground[i] = ground[i - 1];
        }
        ground[0] = 1;
        for(int i = 2 ; i > 0 ; i--){
          ground[i] = ground[i - 1];
        }
        ground[0] = 0;
        score += tmp;
      }
      else if(result == 3){
        int tmp = ground[2] + ground[1] + ground[0];
        for(int i = 2 ; i > 0 ; i--){
          ground[i] = ground[i - 1];
        }
        ground[0] = 1;
        for(int i = 2 ; i > 0 ; i--){
          ground[i] = ground[i - 1];
        }
        ground[0] = 0;
        for(int i = 2 ; i > 0 ; i--){
          ground[i] = ground[i - 1];
        }
        ground[0] = 0;
        score += tmp;
      }
      else if(result == 4){
        int tmp = ground[2] + ground[1] + ground[0] + 1;
        ground = new int[3];
        score += tmp;

      }
      else if(result == 0){
        out++;
        if(out > 2){
          inning++;
          ground = new int[3];
          out = 0;
        }
      }
      playerN = (playerN + 1) % 9;
    }
    ans = Math.max(ans, score);

  }
  public static void main(String[] args) throws IOException {
    BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    N = Integer.parseInt(br.readLine());
    results = new int[N][9];
    for(int i = 0 ; i < N ; i++){
      results[i] = Arrays.stream(br.readLine().split(" ")).mapToInt(Integer::parseInt).toArray();
    }
    permutation(0, new int[8], new boolean[8]);
    System.out.println(ans);
  }
}
